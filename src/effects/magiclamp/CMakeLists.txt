set(magiclamp_SOURCES
    magiclamp.cpp
    main.cpp
)

kconfig_add_kcfg_files(magiclamp_SOURCES
    magiclampconfig.kcfgc
)

    kwineffects

    KF${KF_MAJOR_VERSION}::ConfigGui
)

    LIBRARY_OUTPUT_NAME "magiclamp"
)
